function deleteObject() {
	const ids = [];
	
	const objectIds = document.querySelectorAll("input[name='objectId']");
	
	for(let i = 0; i < objectIds.length; i++) {
		if(objectIds[i].checked == true) {
			ids.push(objectIds[i].value);
		}
	}
	
	if(ids.length == 0) {
		alert("삭제하실 데이터를 선택해주세요");
		return;
	}
	document.querySelector("#deleteForm").submit();
	
}
